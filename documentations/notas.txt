# Servidor HTTPS para xplagiax.ca
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name xplagiax.ca;
    root /var/www/html;
    index index.html index.htm index.nginx-debian.html;

    # Certificados SSL
    ssl_certificate /etc/letsencrypt/live/xplagiax.ca/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/xplagiax.ca/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    # Configuración principal del sitio
    location / {
        try_files $uri $uri/ =404;
    }

    # Redireccionar /adm sin slash final
    location = /adm {
        return 301 $scheme://$server_name/adm/;
    }

    # Proxy para SocketIO específicamente (DEBE estar antes de /adm/)
    location /adm/socket.io/ {
        proxy_pass http://localhost:5001/socket.io/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }

    # Proxy para archivos estáticos - Flask los sirve internamente
    location /adm/static/ {
        proxy_pass http://localhost:5001/static/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # Headers para caching
        expires 1y;
        add_header Cache-Control "public, immutable";

        # Configuración para archivos estáticos
        proxy_buffering on;
        proxy_cache_valid 200 1y;
    }

    location /adm/assets/ {
        proxy_pass http://localhost:5001/assets/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # Headers para caching
        expires 1y;
        add_header Cache-Control "public, immutable";

        # Configuración para archivos estáticos
        proxy_buffering on;
        proxy_cache_valid 200 1y;
    }

    # Proxy principal para el contenedor Docker en /adm
    location /adm/ {
        proxy_pass http://localhost:5001/;
        proxy_ssl_verify off;
        proxy_set_header Host $host;
        proxy_set_header X-Script-Name /adm;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # CRÍTICO: Estos headers le dicen a Flask que está bajo /adm
        proxy_set_header X-Script-Name /adm;
        proxy_set_header X-Forwarded-Prefix /adm;

        proxy_buffering off;
        proxy_http_version 1.1;
        proxy_set_header Connection "";

        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;

        # Configuración para SocketIO
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";

        # Reescribir redirects de Flask
        proxy_redirect http://localhost:5001/ /adm/;
        proxy_redirect http://$host/ /adm/;
        proxy_redirect https://$host/ /adm/;
    }

    # Denegar acceso a archivos .htaccess
    location ~ /\.ht {
        deny all;
    }
}
